Simulation of multi-compartment HH model¶

According to the Kirchhoffs law it must hold that the currents of a compartment $j$ and adjoining compartments $j-1$ and $j+1$ sum up to zero : $0 = I_{j-1} + I_j + I_{j+1}$. Within a compartment $j$ four different currents can be measured . One caused by the capacitor ($C_m \frac{dV'}{dt}$), two caused by the Ion flux of Na and K ($g_{Na,j}(V_j - E_{Na}),g_{K,j}(V_j - E_{K})$) , one caused by leakage ($g_{L}(V_j - E_{L})$)and one potential externel input current $I_e(t)$. Using Kichhoffs current law again you can say that the current of compartment $j$ can be written by: $I_j= C_m \frac{dV}{dt} + g_{L}(V_j - E_{L}) + g_{Na,j}(V_j - E_{Na}) + g_{K,j}(V_j - E_{K}) - I_e(t)$. The currents flowing from adjoining compartments into compartment $j$ can be written by: $I_{j-1} = g_{ax}(V_j - V_{j-1})$ and $I_{j+1} = g_{ax}(V_j - V_{j+1})$. Plugging all together gives us the DEQ:

$$ 0 = C_m \frac{dV}{dt} + g_{L}(V_j - E_{L}) + g_{Na,j}(V_j - E_{Na}) + g_{K,j}(V_j - E_{K}) - I_e(t) + g_{ax}(V_j - V_{j-1}) + g_{ax}(V_j - V_{j+1}) \\ C_m \frac{dV}{dt} = I_e(t) - g_{L}(V_j - E_{L}) - g_{Na,j}(V_j - E_{Na}) - g_{K,j}(V_j - E_{K}) - g_{ax}(V_j - V_{j-1}) - g_{ax}(V_j - V_{j+1}) \\ \frac{dV}{dt} = \frac{I_e(t) - g_{L}(V_j - E_{L}) - g_{Na,j}(V_j - E_{Na}) - g_{K,j}(V_j - E_{K}) - g_{ax}(V_j - V_{j-1}) - g_{ax}(V_j - V_{j+1})}{C_m}$$

with: $g_{Na,j} = \hat{g}_{Na}m_j^3h_j$, $g_{K,j} = \hat{g}_{K}n_j^4$ and the DEQ:

$$\frac{dm}{dt} = \alpha_m(V(j,t)) (1-m) - \beta_m(V(j,t))m \\ \frac{dh}{dt} = \alpha_h(V(j,t)) (1-h) - \beta_h(V(j,t))h \\ \frac{dn}{dt} = \alpha_m(V(j,t)) (1-n) - \beta_n(V(j,t))n$$

In order to derive the voltage at compartment j at time t you can approximate all voltage dependend conductances in this compartment at time t using forward euler method.

$$ m_j(t + \Delta t) = m_j(t) + \Delta t \cdot \frac{dm_j}{dt} \\ h_j(t + \Delta t) = h_j(t) + \Delta t \cdot \frac{dh_j}{dt} \\ n_j(t + \Delta t) = n_j(t) + \Delta t \cdot \frac{dn_j}{dt} \\ g_{Na,j,t} = \hat{g}_{Na}m_j(t)^3h_j(t) \\ g_{K,j,t} = \hat{g}_{K}n_j(t)^4$$

Having approximated these conductances you can derive V(j,t) by using forward euler with:

$$ V(j,t+\Delta t) = V(j,t) + \Delta t \cdot \frac{dV}{dt}$$

def dV_dt(V_i,V_j,V_k,input_function, m, h, n, j, t):
    
    current_V = input_function(j,t) - g_L*(V_j - E_L) - conductance_Na(m,h) * (V_j - E_Na) - conductance_K(n) * (V_j - E_K)
    prev_V = g_ax*(V_j - V_i)
    next_V = g_ax*(V_j - V_k)
    
    return (current_V - prev_V -  next_V)/C_m

def steady_state(V_0):
    time = np.arange(0,500,0.025)
    n = np.zeros(1)
    m = np.zeros(1)
    h = np.zeros(1)
       
    for t in time:
        n_old = n[0]
        delta_n = dn_dt(V_0,n_old)
        n_new = n_old + delta_t * delta_n 
        n[0] = n_new
           
        m_old = m[0]
        delta_m = dm_dt(V_0, m_old)
        m_new = m_old + delta_t * delta_m 
        m[0] = m_new
       
        h_old = h[0]
        delta_h = dh_dt(V_0, h_old)
        h_new = h_old + delta_t * delta_h 
        h[0] = h_new
       
    return m[0],h[0],n[0]

def approximate_model(N, V_0, input_function, duration, delta_t):
    m_0,h_0,n_0 = steady_state(V_0)
    steps = np.arange(0,duration,delta_t)
    
        
    V = np.zeros((N+2, steps.shape[0]))
    m = np.zeros((N+2, steps.shape[0]))
    h = np.zeros((N+2, steps.shape[0]))
    n = np.zeros((N+2, steps.shape[0]))
       
    V[0:N+2,0] = [V_0 for i in range(N+2)]
    m[0:N+2,0] = [m_0 for i in range(N+2)]
    h[0:N+2,0] = [h_0 for i in range(N+2)]
    n[0:N+2,0] = [n_0 for i in range(N+2)]
    
    
       
    for i,t in enumerate(steps[:-1]):
        for j in range(1,N+1):
            V_old = V[j,i]
            m_old = m[j,i]
            h_old = h[j,i]
            n_old = n[j,i]
           
            
           
            delta_m = dm_dt(V_old, m_old)
            m_new = m_old + delta_t * delta_m
            m[j,i+1] = m_new
       
            delta_h = dh_dt(V_old, h_old)
            h_new = h_old + delta_t * delta_h
            h[j,i+1] = h_new       
               
            delta_n = dn_dt(V_old, n_old)
            n_new = n_old + delta_t * delta_n
            n[j,i+1] = n_new
       
               
            delta_V = dV_dt(V[j-1,i],V_old,V[j+1,i],input_function, m_old, h_old, n_old, j, t)
            V_new = V_old + delta_t * delta_V
            V[j,i+1] = V_new
        
        V[0,i+1] = V[1,i+1]
        
       
    
    return V,m,h,n,steps
